unit identity operator

[csharp]
let (~~) (func:'a-> unit) (arg:'a) = (func arg) |> fun () -> arg
[/csharp]

[csharp]
let print2' item = ~~ (printf "%s") item

let print' item = ~~ (Console.WriteLine:string->unit) (item.ToString())
[/csharp]
